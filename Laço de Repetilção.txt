#include <stdio.h>

//Construa um programa em C que leia 10 valores inteiros e positivos e:
//Encontre o maior valor;
//Encontre o menor valor;
//Calcule a média dos números lidos

int main() {
    int a, valor;
    int maior, menor;
    int soma = 0;

    printf("Digite 10 valores inteiros e positivos:\n");
  
    scanf("%d", &valor);
    maior = valor;
    menor = valor;
    soma += valor;

    for (a = 1; a < 10; a++) {
        scanf("%d", &valor);
        soma += valor;
        
        if (valor > maior) {
            maior = valor;
        }

        if (valor < menor) {
            menor = valor;
        }
    }

    float media = (float)soma / 10;

    printf("Maior valor: %d\n", maior);
    printf("Menor valor: %d\n", menor);
    printf("Media: %.2f\n", media);

    return 0;
}

#include <stdio.h>

//Faça um programa em C que conte de 1 a 100 e a cada múltiplo de 10 emita uma mensagem: “Múltiplo de 10”.

int main() {
    int x;

    for (x = 1; x <= 100; x++) {
        printf("%d\n", x);

        if (x % 10 == 0) {
            printf("Múltiplo de 10\n");
        }
    }

    return 0;
}

#include <stdio.h>

//Escreva um programa em C que leia 10 números inteiros e imprima quantos são pares e quantos são ímpares.

int main() {
    int y, valor;
    int pares = 0, impares = 0;

    printf("Digite 10 numeros inteiros:\n");

    for (y = 0; y < 10; y++) {
        scanf("%d", &valor);

        if (valor % 2 == 0) {
            pares++;
        } else {
            impares++;
        }
    }

    printf("Quantidade de numeros pares: %d\n", pares);
    printf("Quantidade de numeros impares: %d\n", impares);

    return 0;
}
